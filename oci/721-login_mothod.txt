//------------------------------------------------------------------------------------------
//
// 패스워드 접속시
//
//------------------------------------------------------------------------------------------

   -. 패스워드 접속허용 
     sudo vi /etc/ssh/sshd_config
       -> PasswordAuthentication yes    

   -. 서비스 다시시작 
     sudo /etc/init.d/ssh restart    


//------------------------------------------------------------------------------------------
//
// 개인키 접속시
//
//------------------------------------------------------------------------------------------
  
   -. 키 접속허용 
     sudo vi /etc/ssh/sshd_config
       -> PasswordAuthentication no

   -. 서비스 다시시작 
     sudo /etc/init.d/ssh restart  

   -. 퍼블릭키 추가 
     echo {퍼블릭키값} > .ssh/authorized_keys
